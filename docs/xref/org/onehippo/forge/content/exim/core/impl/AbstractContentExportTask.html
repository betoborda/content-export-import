34  /**
35   * Abstract content export task implementation class to provide common properties and utility operations.
36   */
37  abstract public class AbstractContentExportTask extends AbstractContentMigrationTask {
38  
39      protected ContentNodeMapper<Node, Item, Value> contentNodeMapper;
40      protected ContentNodeMappingItemFilter<Item> contentNodeMappingItemFilter;
41  
42      /**
43       * Constructs with {@code documentManager}.
44       * @param documentManager {@link DocumentManager} instance
45       */
46      public AbstractContentExportTask(final DocumentManager documentManager) {
47          super(documentManager);
48      }
49  
50      /**
51       * Returns {@link ContentNodeMapper} instance. If not set, returns a default implementation.
52       * @return {@link ContentNodeMapper} instance. If not set, returns a default implementation
53       */
54      public ContentNodeMapper<Node, Item, Value> getContentNodeMapper() {
55          if (contentNodeMapper == null) {
56              contentNodeMapper = new DefaultJcrContentNodeMapper();
57          }
58  
59          return contentNodeMapper;
60      }
61  
62      /**
63       * Sets {@link ContentNodeMapper} instance.
64       * @param contentNodeMapper {@link ContentNodeMapper} instance
65       */
66      public void setContentNodeMapper(ContentNodeMapper<Node, Item, Value> contentNodeMapper) {
67          this.contentNodeMapper = contentNodeMapper;
68      }
69  
70      /**
71       * Returns {@link ContentNodeMappingItemFilter} instance. If not set, returns a default implementation.
72       * @return {@link ContentNodeMappingItemFilter} instance. If not set, returns a default implementation
73       */
74      public ContentNodeMappingItemFilter<Item> getContentNodeMappingItemFilter() {
75          if (contentNodeMappingItemFilter == null) {
76              contentNodeMappingItemFilter = new DefaultHippoJcrItemMappingFilter();
77          }
78  
79          return contentNodeMappingItemFilter;
80      }
81  
82      /**
83       * Sets {@link ContentNodeMappingItemFilter} instance.
84       * @param contentNodeMappingItemFilter {@link ContentNodeMappingItemFilter} instance
85       */
86      public void setContentNodeMappingItemFilter(ContentNodeMappingItemFilter<Item> contentNodeMappingItemFilter) {
87          this.contentNodeMappingItemFilter = contentNodeMappingItemFilter;
88      }
89  
90      /**
91       * Set meta properties such as {@link Constants#META_PROP_NODE_PATH} and {@link Constants#META_PROP_NODE_LOCALIZED_NAME},
92       * which might be helpful when importing back later.
93       * @param contentNode {@link ContentNode} instance to set the meta properties
94       * @param sourceNode the source node from which the meta properties should be extracted
95       * @throws RepositoryException if data cannot be read from the {@code sourceNode} due to repository error
96       */
97      protected void setMetaProperties(final ContentNode contentNode, final Node sourceNode) throws RepositoryException {
98          final Node handle = HippoNodeUtils.getHippoDocumentHandle(sourceNode);
99  
100         if (handle != null) {
101             contentNode.setProperty(Constants.META_PROP_NODE_PATH,
102                     ContentPathUtils.removeIndexNotationInNodePath(handle.getPath()));
103 
104             if (handle instanceof HippoNode) {
105                 contentNode.setProperty(Constants.META_PROP_NODE_LOCALIZED_NAME,
106                         ((HippoNode) handle).getDisplayName());
107             }
108         } else {
109             contentNode.setProperty(Constants.META_PROP_NODE_PATH,
110                     ContentPathUtils.removeIndexNotationInNodePath(sourceNode.getPath()));
111 
112             if (sourceNode instanceof HippoNode) {
113                 contentNode.setProperty(Constants.META_PROP_NODE_LOCALIZED_NAME, ((HippoNode) sourceNode).getDisplayName());
114             }
115         }
116     }
