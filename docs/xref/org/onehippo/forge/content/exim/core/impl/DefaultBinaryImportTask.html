47      /**
48       * {@inheritDoc}
49       */
50      @Override
51      public ContentNodeBindingItemFilter<ContentItem> getContentNodeBindingItemFilter() {
52          if (contentNodeBindingItemFilter == null) {
53              DefaultContentNodeJcrBindingItemFilter filter = new DefaultContentNodeJcrBindingItemFilter();
54              filter.addPropertyPathExclude(Constants.META_PROP_NODE_LOCALIZED_NAME);
55              filter.addPropertyPathExclude(Constants.META_PROP_NODE_PATH);
56              filter.addPropertyPathExclude("hippostdpubwf:*");
57              filter.addPropertyPathExclude("hippo:availability");
58              filter.addPropertyPathExclude("hippo:paths");
59              filter.addPropertyPathExclude("hippo:related");
60              filter.addPropertyPathExclude("hippostd:holder");
61              filter.addPropertyPathExclude("hippostd:state");
62              filter.addPropertyPathExclude("hippostd:stateSummary");
63              contentNodeBindingItemFilter = filter;
64          }
65  
66          return contentNodeBindingItemFilter;
67      }

69      /**
70       * {@inheritDoc}
71       */
72      @Override
73      public String createOrUpdateBinaryFolder(String folderLocation, String primaryTypeName, String[] folderTypes,
74              String[] galleryTypes) throws ContentMigrationException {
75          String folderPath = null;
76  
77          try {
78              final Node folderNode = HippoBinaryNodeUtils.createMissingHippoBinaryFolders(getDocumentManager().getSession(),
79                      folderLocation, primaryTypeName, folderTypes, galleryTypes);
80              getDocumentManager().getSession().save();
81              folderPath = folderNode.getPath();
82          } catch (RepositoryException | WorkflowException e) {
83              try {
84                  getDocumentManager().getSession().refresh(false);
85              } catch (RepositoryException re) {
86                  e.printStackTrace();
87              }
88  
89              throw new ContentMigrationException(e.toString(), e);
90          }
91  
92          return folderPath;
93      }

95      /**
96       * {@inheritDoc}
97       */
98      @Override
99      public String createOrUpdateBinaryFromContentNode(ContentNode contentNode, String primaryTypeName,
100             String folderPath, String name) throws ContentMigrationException {
101         String binaryContentPath = null;
102         Node binaryHandleNode = null;
103 
104         try {
105             if (getCurrentContentMigrationRecord() != null) {
106                 getCurrentContentMigrationRecord().setContentType(primaryTypeName);
107             }
108 
109             if (!getDocumentManager().getSession().nodeExists(folderPath)) {
110                 throw new IllegalArgumentException("Folder doesn't exist at " + folderPath);
111             }
112 
113             final Node folderNode = getDocumentManager().getSession().getNode(folderPath);
114 
115             if (!folderNode.hasNode(name)) {
116                 binaryHandleNode = createBinaryHandleAndVariantNode(primaryTypeName, folderNode, name);
117                 binaryContentPath = binaryHandleNode.getPath();
118             } else {
119                 binaryHandleNode = folderNode.getNode(name);
120                 binaryContentPath = binaryHandleNode.getPath();
121             }
122 
123             if (getCurrentContentMigrationRecord() != null) {
124                 getCurrentContentMigrationRecord().setContentId(binaryHandleNode.getIdentifier());
125             }
126 
127             updateBinaryHandleAndVariantNodeFromBinaryVariantContentNode(binaryHandleNode, contentNode);
128 
129             getDocumentManager().getSession().save();
130         } catch (Exception e) {
131             try {
132                 getDocumentManager().getSession().refresh(false);
133             } catch (RepositoryException re) {
134                 e.printStackTrace();
135             }
136 
137             throw new ContentMigrationException(e.toString(), e);
138         }
139 
140         return binaryContentPath;
141     }

143     /**
144      * Creates a binary handle and variant node and returns the binary handle node.
145      * @param primaryTypeName primary node type name of the binary variant node
146      * @param folderNode folder node
147      * @param name binary node name
148      * @return created binary handle node
149      * @throws DocumentManagerException if it fails to create the binary handle and variant node
150      * @throws RepositoryException if it fails to create the binary handle and variant node
151      */
152     protected Node createBinaryHandleAndVariantNode(String primaryTypeName, Node folderNode, String name)
153             throws DocumentManagerException, RepositoryException {
154         Node handle = folderNode.addNode(name, HippoNodeType.NT_HANDLE);
155         handle.addMixin("mix:referenceable");
156         Node binarySetNode = handle.addNode(name, primaryTypeName);
157         binarySetNode.addMixin("mix:referenceable");
158         binarySetNode.setProperty(HippoNodeType.HIPPO_AVAILABILITY, new String[] { "live", "preview" });
159         return handle;
160     }

162     /**
163      * Update binary handle and variant node from the {@link ContentNode} data containing the binary variant content data.
164      * @param binaryHandleNode binary handle node
165      * @param contentNode {@link ContentNode} data containing the binary variant content data
166      * @throws DocumentManagerException if it fails to update the binary handle and variant node
167      * @throws RepositoryException if it fails to update the binary handle and variant node
168      */
169     protected void updateBinaryHandleAndVariantNodeFromBinaryVariantContentNode(final Node binaryHandleNode,
170             final ContentNode contentNode) throws DocumentManagerException, RepositoryException {
171         getContentNodeBinder().bind(binaryHandleNode.getNode(binaryHandleNode.getName()), contentNode,
172                 getContentNodeBindingItemFilter(), getContentValueConverter());
173     }
