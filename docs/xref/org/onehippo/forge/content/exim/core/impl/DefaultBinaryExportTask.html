16  package org.onehippo.forge.content.exim.core.impl;
17  
18  import javax.jcr.Item;
19  import javax.jcr.Node;
20  import javax.jcr.RepositoryException;
21  
22  import org.onehippo.forge.content.exim.core.BinaryExportTask;
23  import org.onehippo.forge.content.exim.core.ContentMigrationException;
24  import org.onehippo.forge.content.exim.core.DocumentManager;
25  import org.onehippo.forge.content.pojo.mapper.ContentNodeMappingItemFilter;
26  import org.onehippo.forge.content.pojo.mapper.jcr.hippo.DefaultHippoJcrItemMappingFilter;
27  import org.onehippo.forge.content.pojo.model.ContentNode;
28  
29  /**
30   * Default {@link BinaryExportTask} implementation.
31   */
32  public class DefaultBinaryExportTask extends AbstractContentExportTask implements BinaryExportTask {
33  
34      /**
35       * Constructs with {@code documentManager}.
36       * @param documentManager {@link DocumentManager} instance
37       */
38      public DefaultBinaryExportTask(final DocumentManager documentManager) {
39          super(documentManager);
40      }
41  
42      /**
43       * {@inheritDoc}
44       */
45      @Override
46      public ContentNodeMappingItemFilter<Item> getContentNodeMappingItemFilter() {
47          if (contentNodeMappingItemFilter == null) {
48              DefaultHippoJcrItemMappingFilter filter = new DefaultHippoJcrItemMappingFilter();
49              filter.addPropertyPathExclude("hippo:availability");
50              filter.addPropertyPathExclude("hippo:paths");
51              filter.addPropertyPathExclude("hippo:text");
52              contentNodeMappingItemFilter = filter;
53          }
54  
55          return contentNodeMappingItemFilter;
56      }
57  
58      /**
59       * {@inheritDoc}
60       */
61      @Override
62      public ContentNode exportBinarySetToContentNode(Node imageSetOrAssetSetNode) throws ContentMigrationException {
63          ContentNode contentNode = null;
64  
65          try {
66              final Node node = imageSetOrAssetSetNode;
67  
68              if (getCurrentContentMigrationRecord() != null) {
69                  getCurrentContentMigrationRecord().setContentType(node.getPrimaryNodeType().getName());
70              }
71  
72              contentNode = getContentNodeMapper().map(node, getContentNodeMappingItemFilter(), getContentValueConverter());
73              setMetaProperties(contentNode, node);
74          } catch (RepositoryException e) {
75              throw new ContentMigrationException(e.toString(), e);
76          }
77  
78          return contentNode;
79      }
80  
81  }
